# Visualization

# 1. Which Age Group Makes More Claims
df['age']=pd.to_numeric(df['age'],errors='coerce')

bins =[0,25,35,45,55,65,100]
labels=['18-25','26-35','36-45','46-55','56-65','65+']
df['age_group']=pd.cut(df['age'],bins=bins,labels=labels)
age_claim=df.groupby('age_group')['total_claim_amount'].sum().reset_index()
sns.barplot(x='age_group',y='total_claim_amount',data=age_claim,color='orange')
plt.title('Claim Amount by Age Group')
plt.xlabel("Age Group")
plt.ylabel("Total Claim Amount")
plt.ticklabel_format(style='plain',axis='y')
plt.show()

# Conclusion : 36-45 Age group makes more claim

# 2. Which Hospital Generates Highest Claim Amount
hospital_claim = df.groupby('hospital_name')['total_claim_amount'].sum().sort_values(ascending=False).head(10)

hospital_claim.plot(kind='bar',color='green')
plt.title("Top Hospitals by Claim Amount")
plt.xlabel("Hospital")
plt.ylabel("Claim Amount")
plt.ticklabel_format(style='plain',axis='y')
plt.show()

# Conclusion :  Kala, Gandhi And Nayar Hospital Generates Claim Amount

# 3. Which Occupation Category Files The Highest Number Of Insurance  Claims ?
occupation_count = df['occupation_type'].value_counts().reset_index()
occupation_count.columns = ['occupation_type','count']

sns.barplot(x='occupation_type', y='count', data=occupation_count,hue='occupation_type',palette='Set2',legend=False)

plt.title("Number of Claims by Occupation")
plt.xlabel("Occupation_type")
plt.ylabel("Number of Claims")

plt.xticks(rotation=45)

plt.show()

# Conclusion : Unemployed And Student Categories Generate Slightly More Claims

# 4. Does Gender Influence The Number Of Claims?
